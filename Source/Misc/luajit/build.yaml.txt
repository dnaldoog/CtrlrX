jobs:
  build:
    name: Build on ${{ matrix.os }}
    runs-on: ${{ matrix.os }}
    strategy:
      matrix:
        os: [windows-latest, ubuntu-latest] # GitHub's Linux is Ubuntu, similar to Fedora

    steps:
      - name: Checkout Code
        uses: actions/checkout@v4
        with:
          submodules: recursive

      - name: Build LuaJIT (Windows)
        if: runner.os == 'Windows'
        shell: cmd
        run: |
          cd Source\Misc\luajit\src
          call "C:\Program Files\Microsoft Visual Studio\2022\Enterprise\VC\Auxiliary\Build\vcvarsall.bat" x64
          msvcbuild.bat

      - name: Build LuaJIT (Linux)
        if: runner.os == 'Linux'
        run: |
          sudo apt-get install -y libwebkit2gtk-4.1-dev libcurl4-openssl-dev
          cd Source/Misc/luajit
          make -j$(nproc)
